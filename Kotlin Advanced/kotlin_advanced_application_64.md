## 요구사항 분석
다음과 같이 변환하는 함수 ```convert(string)```을 작성하시오.

```kotlin
"11011000101110111011010010011011010010101110001100111010111001"
-> [1, 101, 10, 0, 0, 101, 1, 101, 1, 101, 10, 10, 0, 10, 01, 101, 10, 10, 0, 10, 101, 1, 10, 0, 01, 10, 01, 1, 10, 101, 1, 10, 01]
```

## 코드 및 실행 결과
### 코드
```kotlin
fun convert(string: String): List<String> {
    val result = mutableListOf<String>()
    var current = ""
    var lastPrintedResultSize = 0
    
    // 최종 결과에 추가할 현재 ~ 오른쪽 2칸의 값
    val addToResultCases = listOf("000", "001", "010", "110", "111")
    
    // 해당 인덱스  ~ 오른쪽 2칸을 고려
    string.forEachIndexed { idx, it ->
        current += it.toString()
        
        // 오른쪽 2칸이 비어 있으면 강제로 1로 추가
        val testStr = if (idx >= string.length - 2) {
            string.substring(idx) + "1".repeat(idx + 3 - string.length)
        } else {
            string.substring(idx, idx + 3)
        }
        
        // 현재 칸부터 오른쪽 2칸까지의 값에 따라 최종 결과에 추가 여부 결정
        if (addToResultCases.contains(testStr)) {
            result.add(current)
            current = ""
        }
        
        // 20글자마다 출력
        val isMultiple = idx % 20 == 19
        
        if (idx == string.length - 1 || isMultiple) {
            val subList = result.subList(lastPrintedResultSize, result.size)
            println("${subList.joinToString("")} ${subList}")
            
            if (isMultiple) {
                lastPrintedResultSize = result.size
            }
        }
    }
    
    return result
}

fun main() {
    println("result = ${convert("1111011000101110111011010010011011010010101110001100111010111001")}\n")
    println("result = ${convert("0000000100100011010001010110011110001001101010111100110111101111")}\n")
    println("result = ${convert("1111111011011100101110101001100001110110010101000011001000010000")}\n")
}
```
