## 요구사항 분석
어떤 데이터 분포에 대해, 평균값, 중앙값, 최빈값이 어떤 특성을 띄는지에 따라 다음과 같이 구분한다.
* 대칭 분포 : 평균값, 중앙값, 최빈값이 모두 거의 동일하다.
  * 여기서는 이 세 가지 값들 중 최댓값과 최솟값을 포함하여, 그 사이에 있는 자료가 전체의 20% 이하일 때 대칭 분포라고 한다.
* 왼쪽 꼬리 분포 : 일반적으로 평균값 < 중앙값 < 최빈값이다.
  * 여기서는 평균값이 최빈값보다 작으며, 이들을 포함하여 그 사이에 있는 자료가 전체의 20% 초과일 때 왼쪽 꼬리 분포라고 한다.
* 오른쪽 꼬리 분포 : 일반적으로 최빈값 < 중앙값 < 평균값이다.
  * 여기서는 평균값이 최빈값보다 크며, 이들을 포함하여 그 사이에 있는 자료가 전체의 20% 초과일 때 오른쪽 꼬리 분포라고 한다.

다음의 데이터 클래스로 정의된 각 자료에 대해, 평균값, 중앙값, 최빈값 및 이 값들에 대해 해당 값 이상인 자료의 수, 초과인 자료의 수를 출력하고, 최종적으로 분포의 유형을 출력한다.

### 데이터 클래스
```kotlin
// 자료
data class Data (
    val values: List<Int> // 값들의 목록
)
```

// 자료
data class Data (
    val values: List<Int> // 값들의 목록
)

// 데이터 분석 결과 출력
fun printDataAnalysisResult(d: Data) {
    val vals = d.values
    
    val n = vals.size
    
    // 평균값과 중앙값 찾기
    val avg = vals.sum() / n.toDouble()
    val median = when (n % 2) {
        0 -> (vals.sorted()[n / 2 - 1] + vals.sorted()[n / 2]) / 2.0
        1 -> vals.sorted()[n / 2]
        else -> 0
    }
    
    // 최빈값 찾기
    val countInfo = vals.groupBy { it }.toList()
    var useMedianAsMode = false
   
    val mode = when (countInfo.maxOf { it.second.size }) {
        
        // 최빈값의 빈도가 1인 경우 중앙값을 사용
        1 -> {
            useMedianAsMode = true
            median
        }
        else -> {
            countInfo.find { i0 ->
                i0.second.size == countInfo.maxOf { i1 ->
                    i1.second.size
                }
            }!!.first
        }
    }
    
    println("1. ${countInfo}")
    println("2. ${mode}, 중앙값 이용 = ${useMedianAsMode}\n")
    
    // 각 값 이상, 초과인 자료의 수 계산
    val atLeastAvg = vals.count { it >= avg }
    val aboveAvg = vals.count { it > avg }
    val atLeastMedian = vals.count { it >= median.toDouble() }
    val aboveMedian = vals.count { it > median.toDouble() }
    val atLeastMode = vals.count { it >= mode.toDouble() }
    val aboveMode = vals.count { it > mode.toDouble() }
    
    // 값이 클수록 그 이상인 값의 개수는 작아짐
    val atLeastMinOf3 = listOf(atLeastAvg, atLeastMedian, atLeastMode).max()
    val aboveMaxOf3 = listOf(aboveAvg, aboveMedian, aboveMode).min()
    
    // 분포의 유형 판정
    val type = if (atLeastMinOf3 - aboveMaxOf3 <= n / 5.0) {
        "대칭 분포"
    } else if (avg < mode.toDouble() && atLeastAvg - aboveMode > n / 5.0) {
        "왼쪽 꼬리 분포"
    } else if (avg > mode.toDouble() && atLeastMode - aboveAvg > n / 5.0) {
        "오른쪽 꼬리 분포"
    } else {
        "나도 모르는 분포"
    }
    
    // 데이터 출력
    println("데이터 ${n}개, 데이터 중 대표값의 최솟값 ~ 최댓값 사이: ${atLeastMinOf3 - aboveMaxOf3}개")
    println("평균값: ${avg}, 이상: ${atLeastAvg}개, 초과: ${aboveAvg}개")
    println("중앙값: ${median}, 이상: ${atLeastMedian}개, 초과: ${aboveMedian}개")
    println("최빈값: ${mode}, 이상: ${atLeastMode}개, 초과: ${aboveMode}개")
    println("유형: ${type}\n")
}

fun main() {
    val tempData = listOf<Data> (
        Data(listOf(2, 3, 3, 3, 3, 4, 5, 7, 10, 15)),

        Data(listOf(1, 3, 4, 5, 5, 6, 6, 7, 9, 10)),

    )
    
    tempData.forEach {
        printDataAnalysisResult(it)
    }
}
