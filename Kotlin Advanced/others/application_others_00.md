## 요구사항 분석
1000 이상 9999 이하의 모든 자연수에 대해, 각 구간별 (1000-1099, 1100-1199, ..., 9900-9999)별 소수들에 대해 다음의 통계를 출력하시오.
* 해당 범위 내의 소수의 개수, 최댓값, 최솟값, 평균
* 해당 범위 내의 소수 중 ```4n+1```, ```4n-1``` 꼴의 소수의 개수
* 해당 범위 내의 소수 중 ```6n+1```, ```6n-1``` 꼴의 소수의 개수
* 해당 범위 내의 쌍둥이 소수의 개수 (단, 두 소수가 모두 해당 범위 내에 있어야 한다.)

## 코드
```kotlin
import kotlin.math.*

// 기존 average 함수와 달리 소수점 이하 1자리 출력
val List<Int>.avg: Double get() = (this.sum() * 10 / this.size).toInt() / 10.0

fun isPrime(n: Int): Boolean {
    if (n < 1) {
        return false
    }
    
    val sqroot = sqrt(n.toDouble()).toInt()
    for (i in 2..sqroot) {
        if (n % i == 0) {
            return false
        }
    }
    return true
}

fun main() {
    
    fun isanpb(n: Int, a: Int, b: Int) = n % a == b
    
    // 소수 목록 만들기
    val primeList = (10..99).associateWith { mutableListOf<Int>() }
    println("0. ${primeList}")
    
    // 소수 추가하기
    for (i in 1000..9999) {
        if (isPrime(i)) {
            primeList[i / 100]?.add(i)
        }
    }
    println("1. ${primeList[10]}")
    
    // 각 범위에서의 통계 구하기
    for (rangeKey in 10..99) {
        val primeNumList = primeList[rangeKey]!!
        
        println("\n[ 구간 : ${rangeKey}00 ~ ${rangeKey}99 ]")
        println("2. 소수 개수: ${String.format("%02d", primeNumList.size)}, 평균: ${primeNumList.avg}, 최대: ${primeNumList.max()}, 최소: ${primeNumList.min()}")
    	println("3. 4n+1 개수: ${primeNumList.count { isanpb(it, 4, 1) }}")
        println("4. 4n-1 개수: ${primeNumList.count { isanpb(it, 4, 3) }}")
        println("5. 6n+1 개수: ${primeNumList.count { isanpb(it, 6, 1) }}")
        println("6. 6n-1 개수: ${primeNumList.count { isanpb(it, 6, 5) }}")
        
        val primePairs = primeNumList.zipWithNext().filter { it.second - it.first == 2 }
        println("7. 쌍둥이 소수: ${primePairs}")
        println("8. 쌍둥이 소수 개수: ${primePairs.size}")
    }
}
```
